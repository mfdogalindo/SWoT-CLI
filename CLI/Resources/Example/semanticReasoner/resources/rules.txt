# Declaraciones de prefijos
@prefix rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#> .
@prefix sosa: <http://www.w3.org/ns/sosa/> .
@prefix swot: <http://example.org/swot/property/> .
@prefix inference: <http://example.org/swot/inference/> .
@prefix xsd: <http://www.w3.org/2001/XMLSchema#> .

# Regla para detectar temperaturas altas
[DetectHighTemperature:
  (?observation rdf:type sosa:Observation)
  (?observation sosa:observedProperty swot:temperature)
  (?observation sosa:hasSimpleResult ?temperature)
  greaterThan(?temperature, 31)
  ->
  (?observation inference:temperatureAlert "High")
]

# Regla para detectar temperaturas bajas
[DetectHighTemperature:
  (?observation rdf:type sosa:Observation)
  (?observation sosa:observedProperty swot:temperature)
  (?observation sosa:hasSimpleResult ?temperature)
  lessThan(?temperature, 29)
  ->
  (?observation inference:temperatureAlert "Low")
]

# Regla para clasificar la calidad del aire
[ClassifyAirQuality:
  (?observation rdf:type sosa:Observation)
  (?observation sosa:observedProperty swot:airQuality)
  (?observation sosa:hasSimpleResult ?value)
  greaterThan(?value, 80)
  ->
  (?observation inference:airQualityStatus "Good")
]

# Regla para inferir condiciones de confort
[InferComfortConditions:
  (?obs1 rdf:type sosa:Observation)
  (?obs1 sosa:observedProperty swot:temperature)
  (?obs1 sosa:hasSimpleResult ?temp)
  (?obs2 rdf:type sosa:Observation)
  (?obs2 sosa:observedProperty swot:humidity)
  (?obs2 sosa:hasSimpleResult ?humidity)
  greaterThan(?temp, 20)
  le(?temp, 26)
  greaterThan(?humidity, 40)
  lessThan(?humidity, 60)
  ->
  (?obs1 inference:comfortLevel "Optimal")
]

# Regla para categorizar niveles de humedad
[CategorizeHumidityLevel:
  (?observation rdf:type sosa:Observation)
  (?observation sosa:observedProperty swot:humidity)
  (?observation sosa:hasSimpleResult ?humidity)
  greaterThan(?humidity, 60)
  ->
  (?observation inference:humidityLevel "High")
]

[CategorizeHumidityLevel2:
